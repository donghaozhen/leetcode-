class Solution {
    public int twoCitySchedCost(int[][] costs) {
        //飞往b的代价更小
        Arrays.sort(costs, (a, b) -> (a[1] - a[0]) - (b[1] - b[0]));
        int sum = 0;
        for(int i = 0;i<costs.length/2;i++){
            sum = sum + costs[i][1];
        }
        for(int i = costs.length/2;i<costs.length;i++){
            sum = sum + costs[i][0];
        }
        return sum;
    }
}
