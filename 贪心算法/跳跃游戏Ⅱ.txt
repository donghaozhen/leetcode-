class Solution {
    public int jump(int[] nums) {
        int ans = 0, n = nums.length,cur = 0,next = 0;
        for(int i = 0;i<n;i++){
            if(cur<i){
                ans++;
                cur = next;
            }
            next = Math.max(next,i+nums[i]);
        }
        return ans;
    }
}
