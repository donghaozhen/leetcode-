class Solution {
    public int cuttingBamboo(int bamboo_len) {
        int mod = 1000000007;
        
        if (bamboo_len == 2) return 1;
        if (bamboo_len == 3) return 2;
        
        long res = 1;
        int n = bamboo_len;
        
        // 尽量切 3
        while (n > 4) {
            res = res * 3 % mod;
            n -= 3;
        }
        
        // 最后剩下的是 2/3/4
        res = res * n % mod;
        return (int) res;
    }
}
