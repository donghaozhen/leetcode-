import java.util.*;

class Solution {
    public List<List<String>> findLadders(String beginWord, String endWord, List<String> wordList) {
        Set<String> dict = new HashSet<>(wordList);
        List<List<String>> ans = new ArrayList<>();
        if (!dict.contains(endWord)) {
            return ans;
        }
        Map<String, List<String>> graph = new HashMap<>();
        boolean found = bfs(beginWord, endWord, dict, graph);
        if (found) {
            Deque<String> path = new LinkedList<>();
            dfs(endWord, beginWord, graph, path, ans);
        }
        return ans;
    }

    private boolean bfs(String begin, String end, Set<String> dict, Map<String, List<String>> graph) {
        Set<String> curLevel = new HashSet<>();
        Set<String> nextLevel = new HashSet<>();
        curLevel.add(begin);
        boolean findEnd = false;

        while (!curLevel.isEmpty()) {
            // 当前层全部从字典移除，防止回头走更长路径
            dict.removeAll(curLevel);
            for (String word : curLevel) {
                char[] arr = word.toCharArray();
                for (int i = 0; i < arr.length; i++) {
                    char old = arr[i];
                    for (char c = 'a'; c <= 'z'; c++) {
                        arr[i] = c;
                        String nxt = new String(arr);
                        if (dict.contains(nxt)) {
                            if (nxt.equals(end)) {
                                findEnd = true;
                            }
                            graph.computeIfAbsent(nxt, k -> new ArrayList<>()).add(word);
                            nextLevel.add(nxt);
                        }
                    }
                    arr[i] = old;
                }
            }
            if (findEnd) return true;
            // 层迭代
            curLevel.clear();
            curLevel.addAll(nextLevel);
            nextLevel.clear();
        }
        return false;
    }

    private void dfs(String cur, String target, Map<String, List<String>> graph,
                     Deque<String> path, List<List<String>> ans) {
        path.addFirst(cur);
        if (cur.equals(target)) {
            ans.add(new ArrayList<>(path));
        } else if (graph.containsKey(cur)) {
            for (String prev : graph.get(cur)) {
                dfs(prev, target, graph, path, ans);
            }
        }
        path.removeFirst();
    }
}
