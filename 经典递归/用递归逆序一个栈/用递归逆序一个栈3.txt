import java.util.*;

public class ReverseStack {
    int i = 0;
    public int[] reverseStackRecursively(int[] stack, int top) {
        if(top==0){
            return stack;
        }
        int num = stack[top-1];
        i++;
        reverseStackRecursively(stack, top-1);
        i--;
        stack[i] = num;
        return stack;
    }

}
