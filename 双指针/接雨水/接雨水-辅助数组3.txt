class Solution {
    public int trap(int[] height) {
        int sum = 0;
        int n = height.length;
        int[] leftnums = new int[n];
        int[] rightnums = new int[n];
        leftnums[0] = height[0];
        for(int i = 1;i<n;i++){
            leftnums[i] = Math.max(height[i],leftnums[i-1]);
        }
        rightnums[n-1] = height[n-1];
        for(int i = n-2;i>=0;i--){
            rightnums[i] = Math.max(height[i],rightnums[i+1]);
        }
        for(int i = 1;i<n-1;i++){
            sum = sum + Math.max(0,Math.min(leftnums[i-1],rightnums[i+1])-height[i]);
        }
        return sum;
    }
}
