class Solution {
    public int lastStoneWeightII(int[] stones) {
        int n = stones.length;
        int sum = 0;
        for(int stone : stones){
            sum+=stone;
        }
        int[] dp = new int[sum+1];
        for(int i= 0;i<n;i++){
            for(int j = sum;j>=2*stones[i];j--){
                dp[j] = Math.max(dp[j],dp[j-2*stones[i]]+2*stones[i]);
            }
        }
        return sum - dp[sum]; 
    }
}
